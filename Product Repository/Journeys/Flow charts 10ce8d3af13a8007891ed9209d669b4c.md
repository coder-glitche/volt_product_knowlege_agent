## **1. User Acquisition and Onboarding**

### **1.1. Launching the App and User Signup**

```
@startuml
start
partition "User (FE)" {
    :Launch App;
    :View Signup Page;
    if (Click T&C or Privacy Policy?) then (Yes)
        :Click T&C or Privacy Policy;
        :View T&C or Privacy Policy;
    endif
    :Edit Phone Number;
    :Navigate to OTP Page;
}

partition "Backend (BE)" {
    :Trigger OTP;
}

partition "User (FE)" {
    repeat
        :Enter OTP;
        :Submit OTP;
        partition "Backend (BE)" {
            :Verify OTP;
            if (OTP Valid?) then (No)
                :OTP Invalid;
            endif
        }
        if (OTP Valid?) then (No)
            :Display Error Message;
            :Resend OTP?;
            if (Resend OTP?) then (Yes)
                partition "Backend (BE)" {
                    :Trigger OTP;
                }
            endif
        endif
    repeat while (OTP Invalid)
    :Complete Signup;
    :View Verify Email Page;
    if (Verify Email with Google?) then (Yes)
        :Verify Email with Google;
    else
        :Verify Email with Other Method;
        :View Enter Email Page;
        :Enter Email Address;
    endif
    :Email Verification Result;
}

partition "Backend (BE)" {
    :Create User Context;
    :Update User Email;
}

stop
@enduml

```

## **3. Mutual Fund Portfolio Integration**

### **3.1. Linking MF Portfolio**

```
@startuml
start
partition "User (FE)" {
    :View Check Limit Page;
    :Edit Details if Necessary;
    :Update Portfolio Source;
    :Request MF Portfolio Details;
    :Choose CAMS or KFIN?;
}

partition "User (FE)" #LightBlue {
    if (CAMS Selected?) then (Yes)
        :Request OTP for CAMS MF Fetch;
        partition "Backend (BE)" {
            :Trigger OTP for CAMS MF Fetch;
        }
        :Enter OTP;
        partition "Backend (BE)" {
            :Verify OTP for CAMS MF Fetch;
            if (OTP Valid for CAMS?) then (Yes)
                :OTP Verification Successful;
            else
                :OTP Verification Failed;
            endif
        }
    else
        :Request OTP for KFIN MF Fetch;
        partition "Backend (BE)" {
            :Trigger OTP for KFIN MF Fetch;
        }
        :Enter OTP;
        partition "Backend (BE)" {
            :Verify OTP for KFIN MF Fetch;
            if (OTP Valid for KFIN?) then (Yes)
                :OTP Verification Successful;
            else
                :OTP Verification Failed;
            endif
        }
    endif
}

partition "User (FE)" {
    if (OTP Verification Successful?) then (Yes)
        :MF Portfolio Auth Result;
        :View Success Page;
    else
        :Resend OTP?;
        if (Resend OTP?) then (Yes)
            :Resend OTP;
        endif
    endif
    :Get More Portfolio?;
    if (Get More Portfolio?) then (Yes)
        :Repeat Portfolio Linking;
    endif
}

stop
@enduml
```
